#include<iostream>
using namespace std;
int main()
{
	long k,l,m;
	bool arr[1000001];
	arr[0]=0;
	cin>>k>>l>>m;
	long maxi=0;
	long n[m];
	char ans[m];
	for(int i=0;i<m;i++)
	{
		cin>>n[i];
		maxi=max(maxi,n[i]);
	}
	for(long i=1;i<=maxi;i++)
	{
		if(arr[i-1]==0)
		arr[i]=1;
		else if(i-k>=0)
			if(arr[i-k]==0)
			arr[i]=1;
		else if(i-l>=0)
			if(arr[i-l]==0)
			arr[i]=1;
		else
		arr[i]=0;
	}
	for(int i=0;i<m;i++)
	if(arr[n[i]])
	ans[i]='A';
	else
	ans[i]='B';
	for(int i=0;i<m;i++)
	cout<<ans[i];
	
	
	
	
}
